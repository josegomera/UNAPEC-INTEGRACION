CREATE TRIGGER facturaTr 
ON FaturaDB.dbo.Facturas
AFTER INSERT AS 
BEGIN
	INSERT INTO Triggered_Database.[dbo].Facturas(Cliente, Producto, Vendedor, Valor, TipoTransaccion ) 
	SELECT
	Cliente,
	Producto,
	Vendedor,
	Valor,
	TipoTransaccion 
	FROM
	inserted i
END
